
create or replace TABLE PAY_ZONE.PAY_ZONE_SCHEMA.ACCOUNT_TABLE (
	EMAIL VARCHAR(16777216) NOT NULL,
	ACCOUNT_ID VARCHAR(16777216),
	ACCOUNT_NAME VARCHAR(16777216),
	ACCOUNT_NUMBER VARCHAR(16777216),
	ACCOUNT_TYPE VARCHAR(16777216),
	ACCOUNT_SUB_TYPE VARCHAR(16777216),
	ACCOUNT_BALANCE NUMBER(38,2),
	primary key (EMAIL)
);

create or replace TABLE PAY_ZONE.PAY_ZONE_SCHEMA.CONTRACT_TABLE (
	CONTRACT_ID NUMBER(38,0) NOT NULL autoincrement start 1 increment 1 noorder,
	ISSUER_EMAIL VARCHAR(16777216),
	CONTRACTOR_EMAIL VARCHAR(16777216),
	ISSUE_DATE TIMESTAMP_NTZ(9) DEFAULT CURRENT_TIMESTAMP(),
	STATUS VARCHAR(16777216),
	RESPONSE_DATE TIMESTAMP_NTZ(9),
	HOURLY_PAY FLOAT,
	CONTRACT_START_DATE DATE,
	CONTRACT_END_DATE DATE,
	DAILY_START_TIME TIME(9),
	DAILY_END_TIME TIME(9),
	PAYOUT_INTERVAL VARCHAR(16777216),
	primary key (CONTRACT_ID)
);

create or replace TABLE PAY_ZONE.PAY_ZONE_SCHEMA.TRANSACTIONS (
	TRANSACTION_ID NUMBER(38,0) NOT NULL autoincrement start 1 increment 1 noorder,
	DEBITOR_ACCOUNT VARCHAR(16777216),
	CREDITOR_ACCOUNT VARCHAR(16777216),
	AMOUNT FLOAT,
	CONTRACT_ID NUMBER(38,0),
	TRANSACTION_TIME TIMESTAMP_NTZ(9) DEFAULT CURRENT_TIMESTAMP(),
	primary key (TRANSACTION_ID)
);

create or replace TABLE PAY_ZONE.PAY_ZONE_SCHEMA.USER_TABLE (
	EMAIL VARCHAR(16777216) NOT NULL,
	USER_NAME VARCHAR(16777216),
	UUID VARCHAR(16777216),
	primary key (EMAIL)
);

create or replace TABLE PAY_ZONE.PAY_ZONE_SCHEMA.WORK_LOGS (
	LOG_ID NUMBER(38,0) NOT NULL autoincrement start 1 increment 1 noorder,
	CONTRACT_ID VARCHAR(16777216),
	LOG_TIME TIMESTAMP_NTZ(9) DEFAULT CURRENT_TIMESTAMP(),
	primary key (LOG_ID)
);

create or replace TABLE PAY_ZONE.PAY_ZONE_SCHEMA.ZONE_TABLE (
	ZONE_RECORD_ID NUMBER(38,0) NOT NULL autoincrement start 1 increment 1 noorder,
	CONTRACT_ID NUMBER(38,0),
	LATITUDE FLOAT,
	LONGITUDE FLOAT,
	MARKER_IDX NUMBER(38,0),
	primary key (ZONE_RECORD_ID)
);
